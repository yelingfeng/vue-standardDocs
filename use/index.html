          # 使用

          # 初始化

          yarn install

          # 启动

          // 启动
yarn dev

// 构建
yarn build

// eslint 格式化
yarn lint

          # 插件

          # 开发时插件

              yarn add -dev / -D  prettier

          # 运行时插件

              yarn add -save / -S  lodash
